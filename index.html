    function switchTool(tool, event) {

        currentTool = tool;

        document.querySelectorAll('.tool-btn').forEach(btn => btn.classList.remove('active'));

        event.currentTarget.classList.add('active');

        document.getElementById('lab-title').innerText = tools[tool].title;

        document.getElementById('lab-input').value = "";

        document.getElementById('lab-result').style.display = "none";

    }

    async function runAI() {

        const input = document.getElementById('lab-input').value;

        if (!input) return;

        const btn = document.getElementById('run-btn');

        const loader = document.getElementById('btn-loader');

        const result = document.getElementById('lab-result');

        btn.disabled = true; loader.style.display = "block"; result.style.display = "none";

        try {

            const response = await fetch("https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=", {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                    contents: [{ parts: [{ text: tools[currentTool].prompt.replace('INPUT', input) }] }],

                })

            });

            const data = await response.json();

            result.innerText = data.candidates?.[0]?.content?.parts?.[0]?.text || "Ошибка генерации.";

        } catch (e) {
